💻 API使用示例

# Python示例
import requests

response = requests.post("http://localhost:8000/segment", json={
    "text": "你的文本内容...",
    "threshold": 0.5,
    "method": "auto"
})

result = response.json()
print(f"分段成功，共{len(result['paragraphs'])}段")
